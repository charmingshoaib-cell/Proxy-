            STATE.totalProfit = balAfter - STATE.initialBalance;

            const isLoss = diff < -0.000000001;

                STATE.wins++;
                STATE.winStreak++;
                STATE.streak = 0;
                Strategy.onWin();
                Logger.add(`✅ WIN! +${diff.toFixed(8)} | P/L: ${STATE.totalProfit.toFixed(8)}`, 'win');
                Sound.win();
                if (STATE.winStreak >= 5) Sound.bigWin();
            } else if (isLoss) {
                STATE.losses++;
                STATE.streak++;
                STATE.winStreak = 0;
                Strategy.onLoss();
                Logger.add(`❌ LOSS | Streak: ${STATE.streak} | Next: ${STATE.currentBet.toFixed(8)}`, 'loss');

                // Cooldown Check
                if (STATE.streak >= CONFIG.cooldownOnStreak) {
                    this._enterCooldown();
                    updateUI();
                    return;
                }
            }

            if (!CONFIG.unlimitedMode) {
                if (STATE.totalProfit >= CONFIG.targetProfit) {
                    Logger.add(`🎯 TARGET REACHED (+${STATE.totalProfit.toFixed(8)})`, 'win');
                    this.stop(); return;

                    this.stop(); return;
                }
            }

            updateUI();
            this._timer = setTimeout(() => this._loop(), 500);
        },
